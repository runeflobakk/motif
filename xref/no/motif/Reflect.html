1   package no.motif;
2   
3   import java.lang.annotation.Annotation;
4   import java.lang.reflect.AnnotatedElement;
5   import java.lang.reflect.Member;
6   
7   import no.motif.f.Fn;
8   import no.motif.f.Predicate;
9   
10  /**
11   * Operations for doing reflection.
12   */
13  public final class Reflect {
14  
15  
16      /**
17       * Evaluates if objects are of a certain type (class or interface).
18       *
19       * @param type The type that will satisfy the predicate.
20       */
21      public static Predicate<Object> isA(final Class<?> type) { return new Predicate<Object>() {
22          @Override public boolean $(Object object) { return type.isInstance(object); }}; }
23  
24  
25      /**
26       * Gets the class of an object. This is the functional equivalent of
27       * the {@link Object#getClass()} method, which returns an untyped
28       * {@link Class Class&lt;?&gt;}.
29       */
30      public static final Fn<Object, Class<?>> getClass = type();
31  
32  
33      /**
34       * This is the same as {@link #getClass}, with the additional
35       * possible typing of the yielded {@link Class}es.
36       */
37      public static <T> Fn<T, Class<? extends T>> type() { return new Fn<T, Class<? extends T>>() {
38          @SuppressWarnings("unchecked") @Override public Class<? extends T> $(T object) {
39              return (Class<? extends T>) object.getClass(); }}; }
40  
41  
42      /**
43       * Get the name of a {@link java.lang.reflect.Member}, e.g.
44       * class name, method name, field name.
45       */
46      public static final Fn<Member, String> name = new Fn<Member, String>() {
47          @Override public String $(Member member) { return member.getName(); }};
48  
49  
50      /**
51       * Check if an element (class, method, field, etc) is annotated with a given annotation.
52       *
53       * @param annotation the annotation to check for.
54       */
55      public static Predicate<AnnotatedElement> annotatedWith(final Class<? extends Annotation> annotation) {
56          return new Predicate<AnnotatedElement>() {
57              @Override public boolean $(AnnotatedElement element) { return element.isAnnotationPresent(annotation); }}; }
58  
59  
60  
61      /**
62       * Create instances of classes using the no-arg constructor.
63       * Any checked {@link InstantiationException} or {@link IllegalAccessException}
64       * will be rethrown as <code>RuntimeException</code>.
65       */
66      public static <T> Fn<Class<? extends T>, T> newInstance() {
67          return new Fn<Class<? extends T>, T>() {
68              @Override public T $(Class<? extends T> clazz) {
69                  try { return clazz.newInstance(); }
70                  catch (InstantiationException | IllegalAccessException e) {
71                      throw new RuntimeException("Not able to instantiate " + clazz.getName() +
72                              " because of " + e.getClass().getSimpleName() + ": " + e.getMessage(), e);
73                  }
74              }};
75      }
76  
77  
78      private Reflect() {}
79  }
