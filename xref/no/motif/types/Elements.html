1   package no.motif.types;
2   
3   import no.motif.f.Fn;
4   import no.motif.f.Predicate;
5   import no.motif.single.Elem;
6   import no.motif.single.Optional;
7   
8   /**
9    * This is the main type of Motif's collection API and
10   * encompasses all the operations available to manipulate
11   * and query the the contained elements, as well as
12   * converting back to Java Collection Framework
13   * collections.
14   *
15   *
16   * @param <T> The type of the contained objects.
17   */
18  public interface Elements<T> extends
19      Mappable<T>, Filterable<T>, Appendable<T>, Prependable<T>, SideEffectable<T>, Repeatable<T>,
20      HeadAndTail<T>,
21      YieldsJavaCollection<T>,
22      Existance<T>,
23      Iterable<T>,
24      Reducible<T> {
25  
26      /**
27       * Couple each element with its positional index.
28       *
29       * @return The elements wrapped in {@link Elem}s.
30       */
31      Elements<Elem<T>> indexed();
32  
33  
34      /**
35       * Couple each element with its positional index,
36       * starting from a given index.
37       *
38       * @param startIndex the beginning index to assign to the first element.
39       * @return The elements wrapped in {@link Elem}s.
40       */
41      Elements<Elem<T>> indexedFrom(int startIndex);
42  
43  
44      /**
45       * Take only a maximum amount of elements.
46       *
47       * @param amount the amount to take.
48       * @return the elements
49       */
50      Elements<T> take(int amount);
51  
52  
53      /**
54       * Take elements as long as a given predicate
55       * evaluates to true.
56       *
57       * @param predicate The predicate, when evaluates an element to false,
58       *                  will not include that element, nor any more elements.
59       *
60       * @return the elements
61       */
62      Elements<T> takeWhile(Predicate<? super T> predicate);
63  
64  
65      /**
66       * Take elements until a given predicate evaluates to true.
67       *
68       * @param predicate The predicate, when evaluates an element to true,
69       *                  will not include that element, nor any more elements.
70       * @return the elements
71       */
72      Elements<T> takeUntil(Predicate<? super T> predicate);
73  
74  
75  
76      /**
77       * Gets the last element.
78       */
79      Optional<T> last();
80  
81      @Override
82      Optional<T> head();
83  
84      @Override
85      Elements<T> tail();
86  
87  
88  
89      /**
90       * Join all elements to one <code>String</code>.
91       *
92       * @return the joined string
93       */
94      String join();
95  
96  
97      /**
98       * Join all elements to one <code>String</code> with all
99       * elements joined with the given separator.
100      *
101      * @return the joined string
102      */
103     String join(String separator);
104 
105 
106     @Override <O> Elements<O> map(Fn<? super T, O> function);
107     @Override Elements<T> filter(Predicate<? super T> filter);
108     @Override Elements<T> repeat(int times);
109 
110     <O> Elements<O> flatMap(Fn<? super T, ? extends Iterable<O>> fn);
111 
112     /**
113      * Force-evaluate the current elements, applying all composed functions
114      * on each element. This is useful when needing to iterate elements
115      * several times, especially if the composed functions are expensive to evaluate
116      * and/or not referentially transparent.
117      *
118      * This is an escape-hatch from the lazy design of {@link Elements}.
119      *
120      * @return same type of elements, with all elements evaluated.
121      */
122     Elements<T> eval();
123 
124 
125 }
