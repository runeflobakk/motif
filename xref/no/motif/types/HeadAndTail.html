5   import no.motif.Singular;
6   import no.motif.single.Optional;

8   /**
9    * A recursive type which can be splitted into a head element,
10   * and the remaining tail, which may be splitted further into
11   * head and tail.
12   *
13   * @param <E> The type of objects which may be retrieved from the head element.
14   */
15  public interface HeadAndTail<E> {
16  
17      /**
18       * Gets the head, or first element, if it exists.
19       */
20      Optional<E> head();
21  
22      /**
23       * Gets the tail, i.e. the remainder after the head.
24       *
25       * @return The tail, which may be empty if there is only a head element.
26       * @throws NoSuchElementException if {@link #head()} is {@link Singular#none() none}, so
27       *                                there is no remainder.
28       */
29      HeadAndTail<E> tail();
30  
31  }
