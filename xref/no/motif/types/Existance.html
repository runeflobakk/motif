5   /**
6    * A container type which can evaluate for existance of its
7    * element(s), as well as evaluate if it is empty.
8    *
9    * @param <T> The type of the contained element(s)
10   */
11  public interface Existance<T> {
12  
13      /**
14       * @return <code>true</code> if empty, <code>false</code> otherwise.
15       */
16      boolean isEmpty();
17  
18  
19      /**
20       * Decide if an element exist.
21       *
22       * @param predicate evaluates the element(s).
23       * @return <code>true</code> once the predicate evaluates to true,
24       *         otherwise <code>false</code>.
25       */
26      boolean exists(Predicate<? super T> predicate);
27  
28  }
