13  /**
14   * Defines the methods which bridges Motif back to the
15   * Java Collection Framework.
16   *
17   * @param <T> The type of the contained objects.
18   */
19  public interface YieldsJavaCollection<T> {
20  
21      /**
22       * Get the elements as an immutable {@link List}. This is the
23       * most common way to obtain a regular implementation of a Java
24       * collection.
25       *
26       * If you need more control on the returned {@link Collection}
27       * implementation, use {@link #collectIn(Collection)}, e.g. if
28       * you need a mutable collection.
29       *
30       * @return the elements of the iterable as an unmodifiable list.
31       */
32      List<T> collect();
33  
34  
35      /**
36       * Collects the elements in this iterable in the given collection.
37       * The caller is responsible for supplying an appropriate collection
38       * implementation for what behavior that is expected. Keep in mind that
39       * {@link Set sets} does not allow duplicates, and {@link HashSet many implementations}
40       * also does not maintain the order of the elements. The collection must obviously
41       * support the {@link Collection#add(Object) add(..)} operation. Unless you
42       * have particular needs for the resulting collection, consider using
43       * the {@link #collect()} method instead.
44       *
45       * @param collection The collection to add the elements to.
46       * @return the given collection is returned.
47       */
48      <C extends Collection<T>> C collectIn(C collection);
49  
50  
51  
52      /**
53       * Get a sorted immutable {@link List} of the contents of this iterable.
54       *
55       * @param property The function to obtain the property to sort by of each element.
56       * @return the elements of the iterable as a sorted list.
57       */
58      <P extends Comparable<P>> List<T> sortedBy(Fn<? super T, P> property);
59  
60  
61      /**
62       * Get a sorted immutable {@link List} of the contents of this iterable.
63       *
64       * @param comparator The comparator to use when sorting the elements.
65       *                   Use {@link Iterate#byOrderingOf(Class)} to use the
66       *                   natural ordering of the contained elements.
67       * @return the elements of the iterable as a sorted list.
68       */
69      List<T> sorted(Comparator<? super T> comparator);
70  
71  
72      /**
73       * Get a {@link Map} where the keys are derived by applying the given
74       * {@link Fn} on all elements. Each key points to the list of elements
75       * which resolves to the key.
76       *
77       * @param property the function which resolves the key for elements.
78       * @return the map.
79       */
80      <P> Map<P, List<T>> groupBy(Fn<? super T, P> property);
81  
82  
83      /**
84       * Get a {@link Map} where all keys are derived by applying the given
85       * {@link Fn} on all elements. This is a special case of {@link #groupBy(Fn)}
86       * where the <code>Fn</code> will yield a unique key for each element.
87       * In the event of multiple elements resolves to the same key, this method
88       * <em>will fail with an exception</em>. If this uniqueness
89       * cannot be guarantied, consider using {@link #groupBy(Fn)} instead.
90       *
91       * @param uniqueProperty the function which resolves the unique key for each element
92       * @return the map.
93       */
94      <P> Map<P, T> mapBy(Fn<? super T, P> uniqueProperty);
95  
96  }
