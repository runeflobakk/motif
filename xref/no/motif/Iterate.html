30       * Manipulate CharSequences/Strings as if they were iterables of
31       * Characters.
32       *
33       * @param string The string
34       * @return {@link Elements} characters
35       */
36      public static Elements<Character> on(CharSequence string) {
37          if (string == null) return PreparedIterable.empty();
38          List<Character> charList = new ArrayList<>(string.length());
39          for (char c : string.toString().toCharArray()) charList.add(c);
40          return newInstance(charList);
41      }

85       * Create a {@link java.util.Comparator} from an {@link Fn}, typically to

88       * @param property the {@link Fn} which obtains the comparable property

90       * @return a {@link Comparator} using the value yielded from the given {@link Fn}.

92      public static <T, P extends Comparable<P>> Comparator<T> by(final Fn<T, P> property) {
93          return new Comparator<T>() { @Override public int compare(T first, T second) {
94              return property.$(first).compareTo(property.$(second)); }}; }

134     /**
135      * Using a function which takes an object and returns another object of the same type,
136      * recursively call this function until it yields <code>null</code>, and ultimately yield the
137      * last object.
138      *
139      * @param next The function which takes an object and yields another object of the same type.
140      *             This is typically used for objects implementing a linked list-like structure,
141      *             where each object refers to the next in a chain.
142      *
143      * @return A function which will descend the chain and yield the last object. If the given
144      *         <code>next</code> {@link Fn} never returns <code>null</code>, calling this
145      *         returned function will result in an infinite recursion.
146      */
147     public static <T> Fn<T, T> last(final Fn<? super T, ? extends T> next) {
148         return new Fn<T, T>() {
149         @Override public T $(T value) {
150             return optional(next.$(value)).map(this).getOrElse(value);
151         }};
152     }
