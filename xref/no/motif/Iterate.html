20  /**
21   * Operations for containers, e.g. arrays,
22   * {@link List lists}, {@link Set sets}, or any implementation
23   * of {@link Iterable}. Can also
24   * {@link #on(CharSequence) treat Strings as a container of Characters}.
25   *
26   */
27  public final class Iterate {
28  
29      private static final Elements<?> NONE = new PreparedIterable<>(emptyList());
30  
31      /**
32       * @return An empty <code>Elements</code> container.
33       */
34      @SuppressWarnings("unchecked")
35      public static <T> Elements<T> none() {
36          return (Elements<T>) NONE;
37      }
38  
39  
40      /**
41       * Manipulate CharSequences/Strings as if they were iterables of
42       * Characters.
43       *
44       * @param string The string
45       * @return {@link Elements} characters
46       */
47      public static Elements<Character> on(CharSequence string) {
48          if (string == null) return Iterate.none();
49          List<Character> charList = new ArrayList<>(string.length());
50          for (char c : string.toString().toCharArray()) charList.add(c);
51          return newInstance(charList);
52      }
53  
54  
55      /**
56       * Work with multiple elements.
57       *
58       * @param elements the elements to manipulate as vararg/array.
59       * @return {@link Elements}
60       */
61      @SafeVarargs
62      @SuppressWarnings("varargs")
63      public static <T> Elements<T> on(T ... elements) {
64          return newInstance(elements != null ? asList(elements) : null);
65      }
66  
67  
68      public static <K, V> Elements<Map.Entry<K, V>> on(Map<K, V> map) {
69          return on(map != null ? map.entrySet() : null);
70      }
71  
72  
73      /**
74       * Work with multiple elements.
75       *
76       * @param elements the elements to manipulate as an {@link Iterable}.
77       * @return {@link Elements}
78       */
79      public static <T> Elements<T> on(Iterable<T> elements) {
80          return elements instanceof Elements ? (Elements<T>) elements : newInstance(elements);
81      }

88      private static <T> Elements<T> newInstance(Iterable<T> elements) {
89          if (elements == null || !elements.iterator().hasNext()) return none();
90          else return new PreparedIterable<T>(elements);
91      }

95      /**
96       * Create a {@link java.util.Comparator} from an {@link Fn}, typically to
97       * sort using a {@link Comparable} property of a type.
98       *
99       * @param property the {@link Fn} which obtains the comparable property
100      *
101      * @return a {@link Comparator} using the value yielded from the given {@link Fn}.
102      */
103     public static <T, P extends Comparable<P>> Comparator<T> by(final Fn<T, P> property) {
104         return new Comparator<T>() { @Override public int compare(T first, T second) {
105             return property.$(first).compareTo(property.$(second)); }}; }

108     /**
109      * Convert a type with natural ordering, i.e. implements {@link Comparable}, to
110      * an external {@link Comparator}. This is a little quirk to provide a typesafe
111      * way to sort iterables containing <code>Comparables</code>, instead of risking
112      * a runtime ClassCastException. Pass this to the
113      * {@link no.motif.iter.CollectingIterable#sorted(Comparator) .sorted(..)}
114      * method to get elements as a sorted list.
115      *
116      * @param comparableType The {@link Comparable} type to create a {@link Comparator} from.
117      * @return the <code>Comparator</code>
118      */
119     public static <T extends Comparable<T>> Comparator<T> byOrderingOf(Class<T> comparableType) {
120         return by(NOP.<T>fn()); }

145     /**
146      * Using a function which takes an object and returns another object of the same type,
147      * recursively call this function until it yields <code>null</code>, and ultimately yield the
148      * last object.
149      *
150      * @param next The function which takes an object and yields another object of the same type.
151      *             This is typically used for objects implementing a linked list-like structure,
152      *             where each object refers to the next in a chain.
153      *
154      * @return A function which will descend the chain and yield the last object. If the given
155      *         <code>next</code> {@link Fn} never returns <code>null</code>, calling this
156      *         returned function will result in an infinite recursion.
157      */
158     public static <T> Fn<T, T> last(final Fn<? super T, ? extends T> next) {
159         return new Fn<T, T>() {
160         @Override public T $(T value) {
161             return optional(next.$(value)).map(this).getOrElse(value);
162         }};
163     }

166     /**
167      * A bridge from functions yielding arrays, to functions yielding iterables.
168      *
169      * @param yieldsArray the array-yielding function.
170      * @return an adapted function yielding an iterable of the array yielded from the original function.
171      */
172     public static <I, O> Fn<I, Iterable<O>> toIterable(final Fn<I, O[]> yieldsArray) {
173         return new Fn<I, Iterable<O>>() { @Override public Iterable<O> $(I value) { return on(yieldsArray.$(value)); }}; }
