27      /**
28       * Manipulate CharSequences/Strings as if they were iterables of
29       * Characters.
30       *
31       * @param string The string
32       * @return {@link Elements} characters
33       */
34      public static Elements<Character> on(CharSequence string) {
35          if (string == null) return PreparedIterable.empty();
36          List<Character> charList = new ArrayList<>(string.length());
37          for (char c : string.toString().toCharArray()) charList.add(c);
38          return newInstance(charList);
39      }

56       * Work with multiple elements.

58       * @param elements the elements to manipulate as an {@link Iterable}.

61      public static <T> Elements<T> on(Iterable<T> elements) {
62          return elements instanceof Elements ? (Elements<T>) elements : newInstance(elements);

66      private static <T> Elements<T> newInstance(Iterable<T> elements) {
67          if (elements == null) return PreparedIterable.empty();
68          else return new PreparedIterable<T>(elements);

74       * Create a {@link java.util.Comparator} from an {@link Fn}, typically to
75       * sort using a {@link Comparable} property of a type.
76       *
77       * @param property the {@link Fn} which obtains the comparable property
78       *
79       * @return a {@link Comparator} using the value yielded from the given {@link Fn}.
80       */
81      public static <T, P extends Comparable<P>> Comparator<T> by(final Fn<T, P> property) {
82          return new Comparator<T>() { @Override public int compare(T first, T second) {
83              return property.$(first).compareTo(property.$(second)); }}; }

87       * Convert a type with natural ordering, i.e. implements {@link Comparable}, to
88       * an external {@link Comparator}. This is a little quirk to provide a typesafe
89       * way to sort iterables containing <code>Comparables</code>, instead of risking
90       * a runtime ClassCastException. Pass this to the

93       *
94       * @param comparableType The {@link Comparable} type to create a {@link Comparator} from.
95       * @return the <code>Comparator</code>
96       */
97      public static <T extends Comparable<T>> Comparator<T> byOrderingOf(Class<T> comparableType) {
98          return by(NOP.<T>fn()); }
