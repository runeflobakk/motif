6   /**
7    * Utilities for dealing with exceptions.
8    */
9   public final class Exceptions {
10  
11      /**
12       * If you need to rethrow checked exceptions as RuntimeException, this method will
13       * ensure a sensible RuntimeException being acquired. Typical use of this method is
14       * like this:
15       *
16       * <pre> try {
17       *    someMethod(); // may throw checked Exception
18       * } catch (Exception e) {
19       *    throw asRuntimeException(e);
20       * }</pre>
21       *
22       *
23       * @param e The exception.
24       * @return If the given exception is a RuntimeException, it will be returned as-is
25       *         casted to RuntimeException, or else the exception will be wrapped in a
26       *         new RuntimeException.
27       */
28      public static RuntimeException asRuntimeException(Exception e) {
29          if (e instanceof RuntimeException) {
30              return (RuntimeException) e;
31          } else {
32              RuntimeException rtEx = new RuntimeException(e.getClass().getName() + ": " + e.getMessage(), e);
33              rtEx.setStackTrace(e.getStackTrace());
34              return rtEx;
35          }
36      }
37  
38  
39      /**
40       * Get the {@link Throwable#getMessage() message} of a <code>Throwable</code>.
41       */
42      public static final Fn<Throwable, String> message = new Fn<Throwable, String>() {
43          @Override public String $(Throwable throwable) { return throwable.getMessage(); }};
44  
45  
46      /**
47       * Get the {@link Throwable#getCause() cause} of a <code>Throwable</code>.
48       */
49      public static final Fn<Throwable, Throwable> cause = new Fn<Throwable, Throwable>() {
50          @Override public Throwable $(Throwable throwable) { return throwable.getCause(); }};
51  
52  
53      private Exceptions() {}
54  }
