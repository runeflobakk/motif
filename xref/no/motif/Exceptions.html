5   /**
6    * Utilities for dealing with exceptions.
7    */
8   public final class Exceptions {
9   
10      /**
11       * If you need to rethrow checked exceptions as RuntimeException, this method will
12       * ensure a sensible RuntimeException being acquired. Typical use of this method is
13       * like this:
14       *
15       * <pre> try {
16       *    someMethod(); // may throw checked Exception
17       * } catch (Exception e) {
18       *    throw asRuntimeException(e);
19       * }</pre>
20       *
21       *
22       * @param e The exception.
23       * @return If the given exception is a RuntimeException, it will be returned as-is
24       *         casted to RuntimeException, or else the exception will be wrapped in a
25       *         new RuntimeException.
26       */
27      public static RuntimeException asRuntimeException(Exception e) {
28          if (e instanceof RuntimeException) {
29              return (RuntimeException) e;
30          } else {
31              RuntimeException rtEx = new RuntimeException(e.getClass().getName() + ": " + e.getMessage(), e);
32              rtEx.setStackTrace(e.getStackTrace());
33              return rtEx;
34          }
35      }
36  
37      private Exceptions() {}
38  }
