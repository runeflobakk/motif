12  import no.motif.f.base.FalseIfNull;

14  /**
15   * Integer operations. Most of these operations will not fail on numbers to
16   * large to fit into a {@link java.lang.Integer}, only truncate as necessary.
17   * If operating on larger numbers, consider using the equivalent set of operations
18   * from the {@link Longs} class.
19   */
20  public final class Ints {
21  
22      public static final Predicate<Integer> even = new FalseIfNull<Integer>() {
23          @Override public boolean orElse(Integer i) { return i % 2 == 0; }};
24  
25      public static final Predicate<Integer> odd = not(even);
26  
27      /**
28       * Yields the {@link Number#intValue() integer value} of any number.
29       */
30      public static final Fn<Number, Integer> intValue = when(notNull, new Fn<Number, Integer>() {
31          @Override public Integer $(Number value) { return value.intValue(); }});
32  
33      public static final Fn2<Number, Number, Integer> sum = first(Longs.sum).then(intValue);
34  
35      public static final Fn2<Number, Number, Integer> multiply = first(Longs.multiply).then(intValue);
36  
37      public static final Fn2<Number, Number, Integer> divide = first(Longs.divide).then(intValue);
38  
39      public static final Fn<Number, Integer> doubled = first(Longs.doubled).then(intValue);
40  
41      public static final Fn<Number, Integer> rounded = first(Longs.rounded).then(intValue);
42  
43      public static final Fn<Number, Integer> increment = first(Longs.increment).then(intValue);
44  
45      public static final Fn<Number, Integer> add(int value) { return when(notNull, Apply.partially(sum).of(value)); }
46  
47      public static final Fn<Number, Integer> subtract(int value) { return add(value * -1); }
48  
49      public static final Fn<Number, Integer> multipliedBy(int value) { return when(notNull, Apply.partially(multiply).of(value)); }
50  
51      public static final Fn<Number, Integer> dividedBy(int value) { return when(notNull, Apply.partially(argsReversed(divide)).of(value)); }
52  
53  
54      private Ints() {}
55  }
