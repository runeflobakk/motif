229     /**
230      * Get at most a given amount of the first characters of strings.
231      * If the string is shorter than the amount, the original string is returned.
232      */
233     public static Fn<String, String> first(final int charAmount) {
234         return new Fn<String, String>() {
235             @Override public String $(String s) {
236                 if (s == null) return "";
237                 return (charAmount > s.length()) ? s : s.substring(0, charAmount);

247         return new Fn<String, String>() {
