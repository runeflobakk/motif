1   package no.motif;

3   import static no.motif.Exceptions.asRuntimeException;
4   
5   import java.io.IOException;
6   import java.io.OutputStream;
7   import java.io.PrintStream;
8   
9   import no.motif.f.Do;
10  
11  /**
12   * IO-operations.
13   */
14  public final class IO {
15  
16      /**
17       * {@link PrintStream#println(Object) Print} to a <code>PrintStream</code>,
18       * each call adding a line feed at the end.
19       *
20       * @param out the {@link PrintStream} to print to.
21       * @return The <code>println</code> {@link Do} operation.
22       */
23      public static final <T> Do<T> println(final PrintStream out) {
24          return new Do<T>() { @Override public void with(T value) { out.println(value); }}; }
25  
26  
27      /**
28       * {@link PrintStream#print(Object) Print} to a <code>PrintStream</code>.
29       *
30       * @param out the {@link PrintStream} to print to.
31       * @return The <code>print</code> {@link Do} operation.
32       */
33      public static final <T> Do<T> print(final PrintStream out) {
34          return new Do<T>() { @Override public void with(T value) { out.print(value); }}; }
35  
36  
37      /**
38       * {@link OutputStream#write(int) Write} bytes to an <code>OutputStream</code>.
39       *
40       * @param out the {@link OutputStream} to write to.
41       * @return The <code>write</code> {@link Do} operation.
42       */
43      public static final Do<Byte> writeTo(final OutputStream out) {
44          return new Do<Byte>() { @Override public void with(Byte b) {
45              try { out.write(b); } catch (IOException e) { throw asRuntimeException(e); }
46          }};
47      }
48  
49  
50      private IO() {}
51  }
