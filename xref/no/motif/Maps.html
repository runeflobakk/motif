14      /**
15       * Gets the key of a {@link Entry map entry}.
16       */
17      public static <K> Fn<Entry<K, ?>, K> key() { return new Fn<Map.Entry<K, ?>, K>() {
18          @Override public K $(Entry<K, ?> entry) { return entry.getKey(); }}; }
19  
20      /**
21       * Gets the value of a {@link Entry map entry}.
22       */
23      public static <V> Fn<Entry<?, V>, V> value() { return new Fn<Map.Entry<?, V>, V>() {
24          @Override public V $(Entry<?, V> entry) { return entry.getValue(); }}; }
25  
26  
27      private Maps() {}
28  }
