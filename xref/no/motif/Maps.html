3   import java.util.Map;
4   import java.util.Map.Entry;
5   
6   import no.motif.f.Fn;
7   import no.motif.f.Predicate;
8   import no.motif.maps.CombinedMapsView;

10  /**
11   * Operations on {@link Map maps} and {@link Entry map entries}.
12   */
13  public final class Maps {
14  
15  
16      /**
17       * Gets the key of a {@link Entry map entry}.
18       */
19      public static <K> Fn<Entry<K, ?>, K> key() { return new Fn<Map.Entry<K, ?>, K>() {
20          @Override public K $(Entry<K, ?> entry) { return entry.getKey(); }}; }
21  
22  
23      /**
24       * Gets the value of a {@link Entry map entry}.
25       */
26      public static <V> Fn<Entry<?, V>, V> value() { return new Fn<Map.Entry<?, V>, V>() {
27          @Override public V $(Entry<?, V> entry) { return entry.getValue(); }}; }
28  
29  
30  
31      /**
32       * Evaluate if objects are a key {@link Map#containsKey(Object) contained in}
33       * the given <code>Map</code>.
34       */
35      public static <K> Predicate<K> keyIn(final Map<K, ?> map) { return new Predicate<K>() {
36          @Override public boolean $(K key) { return map.containsKey(key); }}; }
37  
38  
39      /**
40       * Evaluate if objects are a value {@link Map#containsValue(Object) contained in}
41       * the given <code>Map</code>.
42       */
43      public static <V> Predicate<V> valueIn(final Map<?, ? extends V> map) { return new Predicate<V>() {
44          @Override public boolean $(V value) { return map.containsValue(value); }}; }
45  
46  
47      /**
48       * Creates a combined view of two maps, where the values of the first map corresponds
49       * to keys of the second map. This has the implication that any value in the first map not
50       * being a key in the second map will appear as not present in this view. The same applies
51       * to keys in the second map not being a value in the first map, they cannot be retrieved with
52       * this view.
53       * <p>
54       * Similarily, the semantics of the {@link Map#containsKey(Object) containsKey(key)} works transitively.
55       * It returns true <em>only</em> if the given key is contained in the first map, and yields a value
56       * which is contained as a key in the second map. The usual behavior considering possible
57       * <code>null</code>s as keys and values applies as specified in the {@link Map} interface, and is
58       * dependent of the two maps the view is based on.
59       * </p>
60       *
61       * @param first The first map. All or a subset of the keys of this map will be keys in the returned view.
62       * @param second The second map. All or a subset of the values of this map will be values in the returned view.
63       * @return the {@link CombinedMapsView}.
64       */
65      public static <K, I, V> Map<K, V> combine(Map<K, I> first, Map<I, V> second) {
66          return new CombinedMapsView<K, I, V>(first, second);
67      }
68  
69  
70      private Maps() {}
71  }
