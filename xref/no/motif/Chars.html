10  /**
11   * Functions operating on single {@link Character characters}.
12   */
13  public final class Chars {
14  
15  
16      /**
17       * Determines if a character is whitespace.
18       * @see Character#isWhitespace(char)
19       */
20      public static final Predicate<Character> whitespace = new FalseIfNull<Character>() {
21          @Override protected boolean orElse(Character c) { return isWhitespace(c); }};
22  
23  
24      /**
25       * Determines if a character is a digit.
26       * @see Character#isDigit(char)
27       */
28      public static final Predicate<Character> digit = new FalseIfNull<Character>() {
29          @Override protected boolean orElse(Character c) { return isDigit(c); }};
30  
31  
32      /**
33       * Determines if a character is a letter.
34       * @see Character#isLetter(char)
35       */
36      public static final Predicate<Character> letter = new FalseIfNull<Character>() {
37          @Override protected boolean orElse(Character c) { return isLetter(c); }};
38  
39  
40      /**
41       * Determines if a character is a letter or a digit.
42       * @see Character#isLetterOrDigit(char)
43       */
44      public static final Predicate<Character> letterOrDigit = new FalseIfNull<Character>() {
45          @Override protected boolean orElse(Character c) { return isLetterOrDigit(c); }};
46  
47

49  }
