19  /**
20   * An <code>Optional</code> wraps a value that is either defined or
21   * undefined. Usually, undefined means <code>null</code>, and defined means
22   * an object instance, but the distinction between defined and undefined can
23   * be customized with a {@link Predicate}.
24   *
25   * @param <V> The type of the wrapped object.
26   *
27   */
28  public abstract class Optional<V> implements Iterable<V>, Mappable<V>, Serializable {

32       * <h1>*** Not part of the public API! ***</h1>
33       * Factory method for resolving a value to {@link Some} or {@link None}.
34       *
35       * @see Singular#optional(Object)
36       * @see Singular#optional(Predicate, Object)
37       * @see Singular#none()
38       */
39      public static <V> Optional<V> resolve(Predicate<? super V> isPresent, V value) {
40          if (isPresent.$(value)) {
41              return new Some<V>(value);
42          } else {
43              return None.getInstance();
44          }
45      }

49       * Wrapper for a 'defined' value. You never under normal
50       * circumstances refer to this type.
51       *
52       * @param <V> The type of the wrapped object.
53       */
54      public static final class Some<V> extends Optional<V> {

56          private final V value;
57  
58          private Some(V value) {
59              this.value = value;
60          }
61  
62          @Override
63          public final Iterator<V> iterator() {
64              return new SingularIterator<V>(value);
65          }
66  
67          @Override
68          public final boolean isSome() {
69              return true;
70          }
71  
72          @Override
73          public final V getOrElse(V fallback) {
74              return get();
75          }
76  
77          @Override
78          public <O> Optional<O> map(Fn<? super V, O> mapper) {
79              O mapped = mapper.$(this.value);
80              return resolve(notNull, mapped);
81          }
82  
83          @Override
84          public <O> Optional<O> map(Predicate<? super O> isPresent, Fn<? super V, O> mapper) {
85              O mapped = mapper.$(this.value);
86              return resolve(isPresent, mapped);
87          }
88  
89          @Override
90          public <O> PreparedIterable<O> split(Fn<? super V, ? extends Iterable<O>> splitter) {
91              return Iterate.on(splitter.$(value));

146     /**
147      * Obtain the value contained in the <code>Optional</code>. To use
148      * this method one <em>must</em> first determine that the
149      * <code>Optional</code> does in fact hold a value.
150      *
151      * To obtain the wrapped value, it is in general preferable to favor
152      * <em>iterating</em> or using {@link #getOrElse(Object)}, which will never

154      *

156      *         <code>Optional</code> without a defined value.

158      * @see #isSome()
