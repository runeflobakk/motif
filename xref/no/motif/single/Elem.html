3   import java.io.Serializable;
4   import java.util.Objects;
5   
6   import no.motif.f.Fn;

8   /**
9    * A simple container for a value coupled with an index.
10   *
11   * @param <T> The type of the containing value.
12   */
13  public final class Elem<T> implements Serializable {
14  
15      public final int index;
16      public final T value;
17  
18      public static <T> Elem<T> of(int index, T value) {
19          return new Elem<T>(index, value);
20      }
21  
22      private Elem(int index, T value) {
23          this.index = index;
24          this.value = value;
25      }
26  
27      @Override
28      public boolean equals(Object o) {
29          if (o instanceof Elem) {
30              Elem<?> another = (Elem<?>) o;
31              return this.index == another.index && Objects.equals(this.value, another.value);
32          }
33          return false;
34      }
35  
36      @Override
37      public int hashCode() {
38          return Objects.hash(index, value);
39      }
40  
41      @Override
42      public String toString() {
43          return "[" + index + ": " + value + "]";
44      }
45  
46      public static <T> Fn<T, Elem<T>> withIndex(final int index) { return new Fn<T, Elem<T>>() {
47          @Override public Elem<T> $(T value) { return new Elem<T>(index, value); }}; }
48  }
