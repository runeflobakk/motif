15  /**
16   * <em>This is a V</em>, i.e. <code>A&lt;V&gt;</code>, a singular value
17   * of type <code>V</code> which is always
18   * considered as "present". Most of the operations from {@link Optional}
19   * are available, except for the ones implying that the value may not be
20   * present. Unlike <code>Optional</code>, calling {@link #get()} on
21   * <code>A&lt;V&gt;</code> will <em>never</em> throw an exception.
22   *
23   * @param <V>
24   */
25  public interface A<V> extends Iterable<V>, Mappable<V>, Filterable<V>, Appendable<V>, Prependable<V>, Serializable {
26  
27      /**
28       *
29       * @return the contained value. Unlike <code>Optional</code>, calling
30       *         {@link #get()} on <code>A&lt;V&gt;</code> will <em>never</em>
31       *         throw an exception.
32       */
33      V get();
34  
35  
36      /**
37       * Map the contained to another value. This will yield an <code>Optional</code> of
38       * the other type, which is "present" for any non-null value yielded from the
39       * mapping.
40       */
41      @Override
42      <O> Optional<O> map(Fn<? super V, O> mapper);
43  
44      /**
45       * Map the value to another value. This will yield an <code>Optional</code> of
46       * the other type. The given {@link Predicate} decides if the resulting value
47       * is "present".
48       *
49       * @see #map(Fn)
50       */
51      <O> Optional<O> map(Predicate<? super O> isPresent, Fn<? super V, O> mapper);
52  
53  
54      @Override
55      Optional<V> filter(Predicate<? super V> filter);
56  
57  
58      /**
59       * Split the value into multiple values.
60       * <p>
61       * Due to limitations of the type system in Java, it is not possible to overload this
62       * method to accept both functions yielding an {@link Iterable} (as this method does)
63       * as well as functions yielding an array. It is recommended to prefer implementing
64       * {@link Fn functions} yielding Iterables. If implementing an array-yielding
65       * function is most appropriate for your case, and you wish to use the function to split
66       * an optional value using this method, it is possible to adapt the function using
67       * {@link Iterate#toIterable(Fn)}
68       * </p>
69       *
70       * @param splitter a function yielding an iterable.
71       * @return the elements.
72       */
73      <O> Elements<O> split(Fn<? super V, ? extends Iterable<O>> splitter);
74  
75  }
