3   import static java.util.EnumSet.allOf;
4   import static no.motif.Base.first;
5   import static no.motif.Base.is;
6   import static no.motif.Iterate.on;
7   import no.motif.f.Fn;
8   
9   public final class Enums {
10  
11      /**
12       * Gets the {@link Enum#name() name} of enum constants.
13       */
14      public static final Fn<Enum<?>, String> name = new Fn<Enum<?>, String>() {
15          @Override public String $(Enum<?> e) { return e.name(); }};
16  
17  
18      /**
19       * Resolve enums from constant names.
20       *
21       * @param enumType The enum type to resolve constants from.
22       */
23      public static <E extends Enum<E>> FromName<E> to(Class<E> enumType) { return new FromName<>(enumType); }
24  
25  
26      /**
27       * Convert from one enum to another by name.
28       *
29       * @param enumType The target enum type
30       */
31      public static <E extends Enum<E>> Fn<Enum<?>, E> toSameName(Class<E> enumType) {
32          return first(name).then(to(enumType));
33      }
34  
35  
36      /**
37       * {@link Fn} which resolves enum constants from names (<code>String</code>).
38       * Use {@link #or(Enum)} to yield a fallback value for non-existing constants,
39       * instead of throwing an exception.
40       */
41      public static final class FromName<E extends Enum<E>> implements Fn<String, E> {
42  
43          private final Class<E> enumType;
44  
45          private FromName(Class<E> enumType) {
46              this.enumType = enumType;
47          }
48  
49          @Override
50          public E $(String name) {
51              return Enum.valueOf(enumType, name);
52          }
53  
54          public final Fn<String, E> or(final E fallback) {
55              return new Fn<String, E>() { @Override public E $(String e) {
56                  return on(allOf(enumType)).map(name).filter(is(e)).map(to(enumType)).head().orElse(fallback);
57              }};
58          }
59      }
60  
61  
62  
63      private Enums() {}
64  
65  }
