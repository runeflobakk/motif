3   /**
4    * Utilities to adapt how functions are invoked.
5    */
6   public final class Apply {
7   
8   
9       /**
10       * Convert an {@link Fn} to a {@link Do}. This is only usable for <code>Fns</code> which
11       * performs some kind of side effect, and you need to pass it somewhere which expects a
12       * <code>Do</code>.
13       *
14       * @param fn The {@link Fn} to apply as a <code>Do</code>.
15       * @return The resulting {@link Do}
16       */
17      public static final <I> Do<I> justSideEffectOf(final Fn<I, ?> fn) {
18          return new Do<I>() { @Override public void with(I value) { fn.$(value); }}; }
19  
20  
21      /**
22       * Transform a function taking 2 arguments to another function where
23       * the argument order is reversed.
24       *
25       * @param fn2 the function
26       * @return a new {@link Fn2} where the argument order is reversed.
27       */
28      public static final <I1, I2, O> Fn2<I2, I1, O> argsReversed(final Fn2<I1, I2, O> fn2) {
29          return new Fn2<I2, I1, O>() { @Override public O $(I2 second, I1 first) { return fn2.$(first, second); }}; }
30  
31  
32      /**
33       * {@link PartialApplication Partially apply} a {@link Fn2 function taking 2 arguments}.
34       *
35       *
36       * @param  fn2 The function to partially apply.
37       * @return use {@link PartialApplication#of(Object) of()}
38       *         to partially invoke <code>fn2</code> and yield an {@link Fn}
39       *
40       * @see    PartialApplication
41       */
42      public static final <I1, I2, O> PartialApplication<Fn2<I1, I2, O>, I1, Fn<I2, O>> partially(Fn2<I1, I2, O> fn2) { return new PartialApplication<Fn2<I1, I2, O>, I1, Fn<I2,O>>(fn2) {
43          @Override public Fn<I2, O> partiallyApply(final Fn2<I1, I2, O> fn2, final Fn0<I1> v1) { return new Fn<I2, O>() {
44              @Override public O $(I2 v2) { return fn2.$(v1.$(), v2); }}; }}; }
45  
46  
47      /**
48       * {@link PartialApplication Partially apply} a {@link Fn function taking 1 argument}.
49       *
50       * @param fn The function to partially apply.
51       * @return use {@link PartialApplication#of(Object) of()}
52       *         to partially invoke <code>fn</code> and yield an {@link Fn0}
53       *
54       * @see    PartialApplication
55       */
56      public static final <I, O> PartialApplication<Fn<I, O>, I, Fn0<O>> partially(Fn<I, O> fn) { return new PartialApplication<Fn<I, O>, I, Fn0<O>>(fn) {
57          @Override public Fn0<O> partiallyApply(final Fn<I, O> fn, final Fn0<I> v1) { return new Fn0<O>() {
58              @Override public O $() { return fn.$(v1.$()); }}; }}; }
