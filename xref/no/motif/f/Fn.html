6   /**
7    * Function taking a value of type <code>I</code> and
8    * returning a value of type <code>O</code>.
9    *
10   * <pre>
11   * &fnof;(I) &isin; O
12   * </pre>
13   *
14   * <p>This function type is often referred to as a "mapper"
15   * or "transformer", as it takes one value and based on it,
16   * yields another value. The function defines the mapping
17   * between values.</p>
18   *
19   * <p>This function type is the most commonly used one, and is
20   * promoted to be named without an index, whereas the other
21   * function types are named as <code>FnX</code>, where
22   * <code>X</code> denotes how many arguments it takes. I.e.
23   * {@link Fn0}, {@link Fn2}, etc.
24   *
25   *
26   * @param <I> the type of the function argument
27   * @param <O> The type of object the function returns.
28   */
29  public interface Fn<I, O> extends FnType, Serializable {
30  
31      /**
32       * Applies the function.
33       *
34       * @param value The value to apply the function on.
35       * @return the result from the function application.
36       */
37      O $(I value);
38  
39  }
