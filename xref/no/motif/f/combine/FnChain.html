5   /**
6    * Composition of two unary functions, where the result of the
7    * first is passed as argument to the second.
8    * <pre>
9    * g(&fnof;(x))
10   * </pre>
11   *
12   *
13   * @param <I>
14   * @param <T>
15   * @param <O>
16   */
17  public class FnChain<I, T, O> implements Fn<I, O> {
18  
19      private final Fn<I, T> first;
20      private final Fn<? super T, O> second;
21  
22      public FnChain(Fn<I, T> first, Fn<? super T, O> second) {
23          this.first = first;
24          this.second = second;
25      }
26  
27      /**
28       * Append a function to the chain.
29       *
30       * @param next The next function.
31       * @return a new chain with the function appended.
32       */
33      public <N> FnChain<I, O, N> then(Fn<? super O, N> next) {
34          return new FnChain<>(this, next);
35      }
36  
37      @Override
38      public O $(I value) {
39          return second.$(first.$(value));
40      }
41  
42  }
