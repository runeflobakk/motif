3   import static java.util.Arrays.asList;
4   import no.motif.Iterate;
5   import no.motif.f.Fn0;
6   import no.motif.types.Appendable;

8   /**
9    * A disjunction is a composition of several {@link Fn0 Fn0&lt;Boolean&gt;s}
10   * which evaluates to <code>true</code> only when at least <em>one</em> of the
11   * premises yields true.
12   * If none of the <code>Fn0&lt;Boolean&gt;s</code> yields <code>true</code>,
13   * the disjunction yields <code>false</code>.
14   */
15  public class DisjunctionPremise implements Fn0<Boolean> {
16  
17      private final Appendable<Fn0<Boolean>> premises;
18  
19      @SafeVarargs
20      @SuppressWarnings("varargs")
21      public DisjunctionPremise(Fn0<Boolean> ... premises) {
22          this(asList(premises));
23      }
24  
25      public DisjunctionPremise(Iterable<Fn0<Boolean>> premises) {
26          this.premises = Iterate.on(premises);
27      }
28  
29      @Override
30      public Boolean $() {
31          for (Fn0<Boolean> p : premises) if (p.$() == Boolean.TRUE) return true;
32          return false;
33      }
34  
35      public DisjunctionPremise or(Fn0<Boolean> otherPremise) {
36          return new DisjunctionPremise(premises.append(otherPremise));
37      }
38  
39  }
