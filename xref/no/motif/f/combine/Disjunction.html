1   package no.motif.f.combine;

3   import static java.util.Arrays.asList;
4   import static no.motif.Base.equalTo;
5   import no.motif.Iterate;
6   import no.motif.f.Predicate;
7   
8   /**
9    * A disjunction is a composition of several predicates which evaluates
10   * to true when at least <em>one</em> of the predicates yields true.
11   * If all the predicates evaluates to false, the disjunction is also
12   * false.
13   */
14  public class Disjunction<T> implements Predicate<T> {
15  
16      private final no.motif.types.Appendable<Predicate<? super T>> predicates;
17  
18      @SafeVarargs
19      @SuppressWarnings("varargs")
20      public Disjunction(Predicate<? super T> ... predicates) {
21          this(asList(predicates));
22      }
23  
24      public Disjunction(Iterable<Predicate<? super T>> predicates) {
25          this.predicates = Iterate.on(predicates);
26      }
27  
28      @Override
29      public boolean $(T value) {
30          for (Predicate<? super T> p : predicates) if (p.$(value)) return true;
31          return false;
32      }
33  
34  
35      public Disjunction<T> or(T other) {
36          return or(equalTo(other));
37      }
38  
39  
40      public Disjunction<T> or(Predicate<? super T> otherPredicate) {
41          return new Disjunction<T>(predicates.append(otherPredicate));
42      }
43  
44  }
