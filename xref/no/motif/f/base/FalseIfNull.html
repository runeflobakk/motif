5   /**
6    * A base implementation for predicates where the case of <code>null</code>
7    * values is evaluated as false. For every other non-null value, sub-classes
8    * must implement the {@link #orElse(Object)} method to decide if
9    * the value should evaluate to <code>true</code> or <code>false</code>,
10   * but need not to worry about checking if the value is <code>null</code>.
11   */
12  public abstract class FalseIfNull<T> implements Predicate<T> {
13  
14      /**
15       * @param value The value to evaluate, never <code>null</code>.
16       * @return <code>true</code> or <code>false</code>
17       */
18      protected abstract boolean orElse(T value);
19  
20      @Override
21      public boolean $(T value) { return value != null && orElse(value); }
22  
23  }
