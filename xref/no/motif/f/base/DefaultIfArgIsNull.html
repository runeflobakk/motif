5   /**
6    * A convenient base implementation of {@link Fn} which yields a default
7    * value for passed <code>null</code> values. For any non-null argument,
8    * the argument is passed to {@link #orElse(Object)}, which
9    * can be implemented without any regard to possible null-pointers.
10   *
11   *
12   * @see Fn
13   */
14  public abstract class DefaultIfArgIsNull<I, O> implements Fn<I, O> {
15  
16      private O defaultValue;
17  
18      protected DefaultIfArgIsNull(O defaultValue) { this.defaultValue = defaultValue; }
19  
20      @Override
21      public final O $(I value) { return value == null ? defaultValue : orElse(value); }
22  
23  
24      /**
25       * The function implementation.
26       *
27       * @param value the function argument, guarantied not to be <code>null</code>.
28       * @return the function result.
29       */
30      protected abstract O orElse(I value);
31  }
