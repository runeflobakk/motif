5   /**
6    * A convenient base implementation of {@link Fn2} which yields
7    * <code>null</code> if one or both arguments is <code>null</code>.
8    * When both arguments are non-null, they are passed to
9    * {@link #orElse(Object, Object)}, which
10   * can be implemented without any regard to possible null-pointers.
11   *
12   * @see Fn2
13   */
14  public abstract class NullIfEitherArgIsNull<I1, I2, O> implements Fn2<I1, I2, O> {
15  
16      @Override
17      public final O $(I1 first, I2 second) {
18          return (first == null || second == null) ? null : orElse(first, second);
19      }
20  
21      /**
22       * The function implementation.
23       * @param first the first function argument, guarantied not to be <code>null</code>.
24       * @param second the second function argument, guarantied not to be <code>null</code>.
25       * @return the function result.
26       */
27      protected abstract O orElse(I1 first, I2 second);
28  
29  }
