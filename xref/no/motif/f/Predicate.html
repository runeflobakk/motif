3   /**
4    * A predicate is a function taking a value of type <code>T</code>, and evaluates it
5    * as either <code>true</code> or <code>false</code>.
6    *
7    * <pre>
8    * &fnof;(T) &isin; {true, false}
9    * </pre>
10   *
11   * Semantically, a predicate is a specialization
12   * of {@link Fn}, with the return type set to primitive {@link Boolean boolean}.
13   * Since Java does not allow type parameterization with primitive types, and predicates
14   * play such an important role, it has been modeled as its own separate type.
15   *
16   * @param <T> The type of objects the predicate can evaluate.
17   */
18  public interface Predicate<T> {
19      boolean $(T value);
20  
21      class Always<T> implements Predicate<T> {
