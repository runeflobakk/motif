3   import java.io.Serializable;

5   /**
6    * A predicate is a function taking a value of type <code>T</code>, and evaluates it
7    * as either <code>true</code> or <code>false</code>.
8    *
9    * <pre>
10   * &fnof;(T) &isin; {true, false}
11   * </pre>
12   *
13   * Semantically, a predicate is a specialization
14   * of {@link Fn}, with the return type set to primitive {@link Boolean boolean}.
15   * Since Java does not allow type parameterization with primitive types, and predicates
16   * play such an important role, it has been modeled as its own separate type.
17   *
18   * @param <T> The type of objects the predicate can evaluate.
19   */
20  public interface Predicate<T> extends Serializable {
21      boolean $(T value);
22  
23      final class Always<T> implements Predicate<T> {
24  
25          private static final long serialVersionUID = 1L;
26  
27          @SuppressWarnings("unchecked")
28          public static <T> Predicate<T> no() { return (Predicate<T>) FALSE; }
29  
30          @SuppressWarnings("unchecked")
31          public static <T> Predicate<T> yes() { return (Predicate<T>) TRUE; }
32  
33  
34          private static final Predicate<?> FALSE = new Always<Object>(false);
35          private static final Predicate<?> TRUE = new Always<Object>(true);
36  
37          private final boolean constant;
38  
39          private Always(final boolean constant) {
40              this.constant = constant;
41          }
42  
43          @Override
44          public final boolean $(T value) {
45              return constant;
46          }
47      }
48  }
