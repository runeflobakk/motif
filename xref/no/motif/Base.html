35      /**
36       * Compose a predicate from taking the result of a function and apply it to
37       * another predicate.
38       *
39       * <pre>
40       * fn(T) &isin; P
41       * p(P) &isin; {true, false}
42       * <b>&fnof;(T) = p(fn(T)) &isin; {true, false}</b>
43       * </pre>
44       *
45       * <p>
46       * This simple composition enables one to focus on writing {@link Fn functions}
47       * for your particular domain, and reusing basic predicates provided by <em>Motif</em>.
48       * </p>
49       *
50       * @param <P>       The type of the predicate evaluating the function result.
51       * @param <T>       The type of the resulting predicate.
52       * @param fn        The function which result will be evaluated by the predicate. Takes
53       *                  an argument of <code>T</code>
54       * @param predicate Evaluates the result from the function, i.e. a value of <code>P</code>
55       * @return          A predicate of <code>T</code>
56       */
57      public static <T, P> Predicate<T> where(final Fn<T, P> fn, final Predicate<? super P> predicate) {
58          return new Where<T, P>(fn, predicate); }
