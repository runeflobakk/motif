41      /**
42       * Compose a predicate from taking the result of a function and apply it to
43       * another predicate.
44       *
45       * <pre>
46       * fn(T) &isin; P
47       * p(P) &isin; {true, false}
48       * <b>&fnof;(T) = p(fn(T)) &isin; {true, false}</b>
49       * </pre>
50       *
51       * <p>
52       * This simple composition enables one to focus on writing {@link Fn functions}
53       * for your particular domain, and reusing basic predicates provided by <em>Motif</em>.
54       * </p>
55       *
56       * @param <P>       The type of the predicate evaluating the function result.
57       * @param <T>       The type of the resulting predicate.
58       * @param fn        The function which result will be evaluated by the predicate. Takes
59       *                  an argument of <code>T</code>
60       * @param predicate Evaluates the result from the function, i.e. a value of <code>P</code>
61       * @return          A predicate of <code>T</code>
62       */
63      public static <T, P> Predicate<T> where(final Fn<T, P> fn, final Predicate<? super P> predicate) {
64          return new Where<T, P>(fn, predicate); }

129     /**
130      * Evaluate if all elements satisfies a predicate, i.e. the <em>Universal quantifier function</em>.
131      * Note that the predicate
132      *
133      * @param predicate The predicate which will evaluate all elements.
134      * @return a predicate which evaluates to <code>true</code> if all elements evaluates
135      *         to true by the given predicate, false if any evaluates to <code>false</code>.
136      */
137     public static <E, I extends Iterable<E>> Predicate<I> all(final Predicate<? super E> predicate) {
138         return new Predicate<I>() { @Override public boolean $(I iterable) {
139             return on(iterable).filter(not(predicate)).isEmpty(); }};}

142     /**
143      * Evaluate if an element exists, i.e. the <em>Existential quantifier function</em>.
144      *
145      * @param element The existance-deciding predicate.
146      * @return a predicate which evaluates to <code>true</code> if any element evaluates
147      *         to true by the given predicate, <code>false</code> otherwise.
148      */
149     public static <E, I extends Iterable<E>> Predicate<I> exists(final Predicate<? super E> element) {
150         return new Predicate<I>() { @Override public boolean $(I iterable) {
151             return on(iterable).exists(element); }};}

171      * Predicate evaluating if values are less than a given value.
172      *
173      * @param value the right operand, i.e. the predicate will evaluate
174      *              if values are less than this value.
175      * @return the "<em>&lt; value</em>" predicate

178         return new Predicate<T>() { @Override public boolean $(T candidate) { return candidate.compareTo(value) < 0; }}; }

181     /**
182      * Predicate evaluating if values are equal to or less than a given value.
183      *
184      * @param value the right operand, i.e. the predicate will evaluate
185      *              if values are equal to or less than this value.
186      * @return the "<em>=&lt; value</em>" predicate
187      */
188     public static <T extends Comparable<? super T>> Predicate<T> equalOrLessThan(T value) {
189         return either(equalTo(value)).or(lessThan(value)); }

199      */

204     /**
205      * Predicate evaluating if values are equal to or greater than a given value.
206      *
207      * @param value the right operand, i.e. the predicate will evaluate
208      *              if values are equal to or greater than this value.
209      * @return the "<em>&gt;= value</em>" predicate

269     /**
270      * Creates a "guarded" {@link Fn} using a {@link Predicate} to evaluate if the
271      * argument should be passed to the wrapped <code>Fn</code>. The function created
272      * by this method is essentially this ternary expression:
273      * <pre> condition(v) ? f(v) : null;</pre>
274      * To fall back to another value than <code>null</code>, call either {@link When#orElse(Fn) .orElse(Fn)}
275      * or {@link When#orElse(Object) .orElse(other)} on the returned object.
276      *
277      * @param condition Only when this evaluates to <code>true</code> will the function argument
278      *                  be applied to the given {@link Fn}.
279      * @param fn        The {@link Fn} to apply if <code>condition</code> evaluates to <code>true</code>
280      */
281     public static <I, O> When<I, O> when(Predicate<? super I> condition, Fn<I, O> fn) {
282         return new When<>(condition, fn);
283     }

286     /**
287      * Compose a chain of several functions into one {@link Fn}, where the result of
288      * each function is passed to its successor, and the last will yield the
289      * actual result of the chain. Use {@link FnChain#then(Fn) .then(Fn)} to append
290      * functions to the chain.
291      *
292      * <p><strong>Note:</strong> The chain does no internal inspection of the intermediate
293      * results passed to the next function, which means that any function which may
294      * return <code>null</code>, <em>must</em> have a <code>null</code>-safe successor
295      * function.
296      *
297      * @param fn The first function in the chain.
298      * @return the given function as the first function in a chain.
299      */
300     public static final <I, O> FnChain<I, O, O> first(Fn<I, O> fn) {
301         return new FnChain<>(fn, NOP.<O>fn());
302     }
