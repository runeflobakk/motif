7   /**
8    * A few no-operation implementations.
9    */
10  public final class NOP {
11  
12  
13      /**
14       * A {@link Runnable} which does nothing and its {@link Runnable#run() run}-method
15       * immediately returns.
16       */
17      public static final Runnable runnable = new Runnable() { @Override public void run() { }};
18  
19      /**
20       * The NOP for {@link Fn} simply yields any given argument as its result.
21       */
22      @SuppressWarnings("unchecked")
23      public static <V> Fn<V, V> fn() { return (Fn<V, V>) fn; }
24  
25      /**
26       * The NOP for {@link Do} does nothing.
27       */
28      public static final Do<Object> doNothing = new Do<Object>() { @Override public void with(Object value) { }};
29  
30  
31      @SuppressWarnings("rawtypes")
32      private static final Fn<?, ?> fn = new Fn() { @Override public Object $(Object value) { return value; }};
33  
34  
35      private NOP() {}
36  }
