3   /**
4    * Base implementation for iterating over elements retrieved using an
5    * incrementing index, where the start and end indexes are known beforehand.
6    * E.g. characters of a String, or elements of an array.
7    */
8   public abstract class PreIndexedContentIterator<T> extends ReadOnlyIterator<T> {
9   
10      private final int endIndex;
11      private int next;
12  
13      /**
14       * @param endIndex The end index, exclusively.
15       */
16      public PreIndexedContentIterator(int endIndex) {
17          this(0, endIndex);
18      }
19  
20      /**
21       * @param startIndex The first index, inclusively
22       * @param endIndex The end index, exclusively.
23       */
24      public PreIndexedContentIterator(int startIndex, int endIndex) {
25          if (endIndex < startIndex) throw new IllegalArgumentException(
26              "endIndex must be same or more than startIndex. startIndex=" + startIndex + ", endIndex=" + endIndex);
27          this.next = startIndex;
28          this.endIndex = endIndex;
29      }
30  
31      @Override
32      public final boolean hasNext() {
33          return next < endIndex;
34      }
35  
36      @Override
37      public final T next() {
38          if (hasNext()) return elementAt(next++);
39          else throw new IndexOutOfBoundsException(String.valueOf(next));
40      }
41  
42      /**
43       *
44       * @param index The index of the element to yield. Guarantied to never
45       *              be out of bounds with respect to the start and end indexes.
46       *              E.g. if the given start and end indexes are 0 and 4, this
47       *              method will be called for each of the indexes 0, 1, 2, and 3.
48       *
49       * @return The element for the given index.
50       */
51      protected abstract T elementAt(int index);
52  }
