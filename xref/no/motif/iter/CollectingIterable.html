16  /**
17   * This iterable offers operations which requires collecting (i.e.
18   * iterating) the containing elements of the iterable. In
19   * particular, this class holds the methods which "bridges" back
20   * to the non-lazy Java Collection Framework classes.
21   *
22   * @param <T> The type of elements in this iterable.
23   */
24  public abstract class CollectingIterable<T> implements Iterable<T>, Reducible<T>, Serializable {

26      @Override
27      public final <O> O reduce(O unit, Fn2<? super O, ? super T, ? extends O> reducer) {
28          O reduced = unit;
29          for (T element : this) reduced = reducer.$(reduced, element);
30          return reduced;
31      }

33      /**
34       * @return the contents of the iterable as an unmodifiable list.
35       */
36      public final List<T> collect() {
37          return unmodifiableList(collectIn(new ArrayList<T>()));
38      }

40      /**
41       * Collects the elements in this iterable in the given collection.
42       * The caller is responsible for supplying an appropriate collection
43       * implementation for what behavior that is expected. Keep in mind that
44       * {@link Set sets} does not allow duplicates, and {@link HashSet many implementations}
45       * also does not maintain the order of the elements. The collection must obviously
46       * support the {@link Collection#add(Object) add(..)} operation. Unless you
47       * have particular needs for the resulting collection, consider using
48       * the {@link #collect()} method instead.
49       *
50       * @param collection The collection to add the elements to.
51       * @return the given collection is returned.
52       */
53      public final <C extends Collection<T>> C collectIn(C collection) {
54          for (T t : this) {
55              collection.add(t);
56          }
57          return collection;
58      }

76      public boolean exists(Predicate<? super T> predicate) {
77          for (T t : this) if (predicate.$(t)) return true;
