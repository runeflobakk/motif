5   import java.io.Serializable;
6   import java.util.Iterator;

8   import no.motif.f.Fn;
9   import no.motif.single.Optional;
10  
11  /**
12   * Iterable where its elements are computed from an initial element.
13   *
14   * @param <T> The type of object this iterable yields.
15   */
16  final class SuccessorIterable<T, F extends T> implements Iterable<T>, Serializable {
17  
18      private final Optional<F> firstElement;
19      private final Fn<? super T,? extends T> successor;
20  
21      SuccessorIterable(F firstElement, Fn<? super T, ? extends T> successor) {
22          this.firstElement = optional(firstElement);
23          this.successor = successor;
24      }
25  
26      @Override
27      public Iterator<T> iterator() {
28          return new SimpleIterator<T>() {
29              Optional<? extends T> next = firstElement;
30              @Override
31              protected Optional<? extends T> nextIfAvailable() {
32                  Optional<? extends T> toReturn = next;
33                  next = next.map(successor);
34                  return toReturn;
35              }};
36      }
37  
38  }
