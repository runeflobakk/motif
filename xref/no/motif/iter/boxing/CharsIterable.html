8   public final class CharsIterable implements Iterable<Character>, Serializable {
9   
10      private final char[] chars;
11  
12      public CharsIterable(char[] chars) { this.chars = chars; }
13  
14      @Override
15      public final Iterator<Character> iterator() {
16          return new PreIndexedContentIterator<Character>(chars.length) {
17              @Override protected Character elementAt(int index) { return chars[index]; }};
18      }
