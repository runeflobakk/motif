433     @Test
434     public void extractAllStringsOccurringBetweenGivenStrings() {
435         assertThat(allBetween(null, null).$("<td>1</td> <td>2</td> <td>3</td>"), emptyIterable());
436         assertThat(allBetween(null, ")").$("(1) (2) (3)"), emptyIterable());
437         assertThat(allBetween("(", null).$("(1) (2) (3)"), emptyIterable());
438         assertThat(allBetween("<td>", "</td>").$("<td>1</td> <td>2</td> <td>3</td>"), contains("1", "2", "3"));
439         assertThat(allBetween("(", ")").$("(1) (2) (3)"), contains("1", "2", "3"));
440         assertThat(allBetween("(", ") ").$("(1) (2) (3)"), contains("1", "2"));
441         assertThat(allBetween("(", "]").$("(1) (2)"), emptyIterable());
442         assertThat(allBetween("", ")").$("(1) (2)"), contains("(1", " (2"));
443         assertThat(allBetween("(", "").$("(1) (2)"), contains("", ""));
