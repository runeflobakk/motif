52      @SuppressWarnings("unchecked")
53      public void createInstances() {

57      @Test(expected = InstantiationException.class)
58      public void createInstanceThrowsException() throws Throwable {
59          try {
60              Reflect.<Integer>newInstance().$(Integer.class);
61          } catch (RuntimeException e) {
62              throw e.getCause();
63          }
64      }

67      @Test
68      public void pickOutObjectsOfCertainSubtype() {
69          List<A> onlyBs = on(new A(), new B()).filter(Reflect.isA(B.class)).collect();
70          assertThat(onlyBs, hasSize(1));
71          assertThat((B) onlyBs.get(0), isA(B.class));
72      }
