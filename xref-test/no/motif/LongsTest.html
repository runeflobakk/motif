18      @Test public void longValueIsNullsafe() {
19          assertThat(longValue.$(null), nullValue());
20      }
21  
22      @Test public void longValue() {

24      }
25  
26      @Test public void rounding() {
27          assertThat(rounded.$(3.14), is(3L));
28          assertThat(rounded.$(4.5), is(5L));
29      }
30  
31      @Test public void subtractValue() {
32          assertThat(on(4, 3).map(subtract(2)), contains(2L, 1L));
33          assertThat(on(4, 3).map(subtract(-2)), contains(6L, 5L));
34      }
35  
36      @Test public void multiplyValue() {
37          assertThat(on(4, 3).map(multipliedBy(3)), contains(12L, 9L));
38      }
39  
40      @Test public void divideValue() {
41          assertThat(on(20, 13).map(dividedBy(4)), contains(5L, 3L));
42      }
