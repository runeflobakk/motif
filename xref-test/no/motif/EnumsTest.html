21  @RunWith(Theories.class)
22  public class EnumsTest {
23  
24      enum A { X, A, B, C, D, E, F, DEFAULT }
25      enum B { X }
26  
27      @Test
28      public void mapBetweenUnrelatedEnumsByName() {
29          assertThat(the(A.X).map(toSameName(B.class)), contains(B.X));
30      }
31  
32  
33      @Theory
34      public void resolveGivenNameAsEnum(@ForAll A a) {
35          assumeThat(a, not(A.DEFAULT));
36          assertThat(the(a.name()).map(to(A.class).or(A.DEFAULT)), contains(a));
37      }
38  
39  
40      @Theory
41      public void resolveToFallbackValue(@ForAll String unknownName) {
42          assumeThat(unknownName, not(isIn(on(A.values()).map(name).collect())));
43          assertThat(the(unknownName).map(to(A.class).or(A.DEFAULT)), contains(A.DEFAULT));
44      }
