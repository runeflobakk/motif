12  public class CyclingIterableTest {
13  
14  
15      @Test
16      public void repeatingNoElementsYieldsNoElementsForAnyRepeatAmount() {
17          assertThat(none().repeat(10), emptyIterable());
18          assertThat(none().repeat(0), emptyIterable());
19          assertThat(none().repeat(-1), emptyIterable());
20      }
21  
22      @Test(expected = IllegalArgumentException.class)
23      public void repeatingSomeElementsANegativeAmountIsNotValid() {
24          on(1, 2, 3).repeat(-4);
25      }
26  
27      @Test
28      public void repeatingZeroTimesYieldsNoElements() {
29          assertThat(on(new SuccessorIterable<>(0, increment)).repeat(0), emptyIterable());
30      }
31  
32      @Test
33      public void repeatingOneTimeYieldsTheSameElements() {
34          assertThat(on("a", "b", "c").repeat(1), contains("a", "b", "c"));
35      }
36  
37      @Test
38      public void repeatingWillCycleTheIterableAGivenAmountOfTimes() {
39          assertThat(on("x", "y").repeat(3), contains("x", "y", "x", "y", "x", "y"));
40      }
