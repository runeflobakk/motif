20  @RunWith(MockitoJUnitRunner.class)
21  public class SideEffectableTest {
22  
23      @Mock
24      private PrintStream out;
25  
26      @Test
27      public void doSideeffectOperation() {
28  
29          on("a", "b", "c").each(print(out));
30          on("1", "2").each(println(out));
31  
32          ArgumentCaptor<String> printCaptor = ArgumentCaptor.forClass(String.class);
33          verify(out, times(3)).print((Object) printCaptor.capture());
34          assertThat(printCaptor.getAllValues(), contains("a", "b", "c"));
35  
36          ArgumentCaptor<String> printlnCaptor = ArgumentCaptor.forClass(String.class);
37          verify(out, times(2)).println((Object) printlnCaptor.capture());
38          assertThat(printlnCaptor.getAllValues(), contains("1", "2"));
39  
40          verifyNoMoreInteractions(out);
41      }
42  }
