120     public void neverContainsValueWhichIsNotInSecondMap(
121             @ForAll(sampleSize=10) LinkedHashMap<String, Integer> first,
122             @ForAll(sampleSize=10) List<Character> valuesInSecond,
123             @ForAll(sampleSize=10) Character valueNotInSecond) {
124 
125         assumeThat(valuesInSecond, not(hasItem(valueNotInSecond)));
126 
127         Queue<Integer> valuesInFirst = new LinkedList<>(first.values());
128         Map<Integer, Character> second = new LinkedHashMap<>();
129         for (Character value : valuesInSecond) {
130             if (valuesInFirst.isEmpty()) break;
131             second.put(valuesInFirst.poll(), value);
132         }
133         Map<String, Character> combined = combine(first, second);
134         assertThat(combined.values(), containsInAnyOrder(on(valuesInSecond).take(first.size()).collect().toArray()));
135         assertThat(combined, not(hasValue(valueNotInSecond)));
136     }

189     public void removingFromEmptyCombinedMapDoesNotRemoveFromUnderlyingMaps(
190             @ForAll(sampleSize=2) Object nonExistingKey,
191             @ForAll(sampleSize=2) Object key1,
192             @ForAll(sampleSize=4) Object value1,
193             @ForAll(sampleSize=4) Object key2,
194             @ForAll(sampleSize=2) Object value2) {
195         assumeThat(value1, not(key2));
196 
197         Map<Object, Object> first = new HashMap<>();
198         first.put(key1, value1);
199         Map<Object, Object> second = new HashMap<>();
200         second.put(key2, value2);
201         Map<?, ?> combined = combine(first, second);
202 
203         assertNull(combined.remove(key1));
204         assertNull(combined.remove(nonExistingKey));
205         combined.clear();
206         assertThat(combined.size(), is(0));
207 
208         assertThat(first.get(key1), is(value1));
209         assertThat(first.size(), is(1));
210         assertThat(second.get(key2), is(value2));
211         assertThat(second.size(), is(1));
212 
213 
214         Object x = "x";
215         Object y = "y";
216         Object z = "z";
217         assumeThat(first, not(hasKey(x)));
218         assumeThat(second, not(hasKey(y)));
219         first.put(x, y);
220         second.put(y, z);
221         assertThat(combined.size(), is(1));
222         assertThat(first.size(), is(2));
223         assertThat(second.size(), is(2));

318     @Test
319     public void unsupportedOperationsBecomeSupportedAfterAMapViewOfFnIsCleared() {
320         Map<String, Do<Map<String, String>>> unsupported = new HashMap<>();
321         unsupported.put("containsValue",
322                 new Do<Map<String, String>>() { @Override public void with(Map<String, String> m) {m.containsValue("a"); }});
323         unsupported.put("entrySet",
324                 new Do<Map<String, String>>() { @Override public void with(Map<String, String> m) {m.entrySet(); }});
325         unsupported.put("keySet",
326                 new Do<Map<String, String>>() { @Override public void with(Map<String, String> m) {m.keySet(); }});
327         unsupported.put("values",
328                 new Do<Map<String, String>>() { @Override public void with(Map<String, String> m) {m.values(); }});
329         unsupported.put("size",
330                 new Do<Map<String, String>>() { @Override public void with(Map<String, String> m) {m.size(); }});
331         unsupported.put("isEmpty",
332                 new Do<Map<String, String>>() { @Override public void with(Map<String, String> m) {m.isEmpty(); }});
333 
334         for (Entry<String, Do<Map<String, String>>> operation : unsupported.entrySet()) {
335             assertUnsupportedUntilCleared(asMap(upperCased), operation);
336         }
337     }
338 
339     private <K, V> void assertUnsupportedUntilCleared(Map<K, V> map, Entry<String, Do<Map<K, V>>> operation) {
340         try {
341             operation.getValue().with(map);
342         } catch (NotPossibleOnMapViewOfFn e) {
343             map.clear();
344             operation.getValue().with(map);
345             return;
346         }
347         fail(operation.getKey() + " should throw exception");
348     }
