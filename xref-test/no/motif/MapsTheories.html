49  
50          Map<Object, Object> map = combinedMap(key, valueKey, valueKey, value);

91      @Theory
92      public void neverContainsValueWhichIsNotInSecondMap(
93              @ForAll(sampleSize=10) Map<String, Integer> first,
94              @ForAll(sampleSize=10) List<Character> valuesInSecond,
95              @ForAll(sampleSize=10) Character valueNotInSecond) {
96  
97          assumeThat(valuesInSecond, not(hasItem(valueNotInSecond)));
98  
99          Queue<Integer> valuesInFirst = new LinkedList<>(first.values());
100         Map<Integer, Character> second = new HashMap<>();
101         for (Character value : valuesInSecond) {
102             if (valuesInFirst.isEmpty()) break;
103             second.put(valuesInFirst.poll(), value);
104         }
105         Map<String, Character> combined = combine(first, second);
106         assertThat(combined.values(), containsInAnyOrder(on(valuesInSecond).take(first.size()).collect().toArray()));
107         assertThat(combined, not(hasValue(valueNotInSecond)));
108     }

160     @Theory
161     public void removingFromEmptyCombinedMapDoesNotRemoveFromUnderlyingMaps(
162             @ForAll(sampleSize=2) Object nonExistingKey,
163             @ForAll(sampleSize=2) Object key1,
164             @ForAll(sampleSize=4) Object value1,
165             @ForAll(sampleSize=4) Object key2,
166             @ForAll(sampleSize=2) Object value2) {
167         assumeThat(value1, not(key2));
168 
169         Map<Object, Object> first = new HashMap<>();
170         first.put(key1, value1);
171         Map<Object, Object> second = new HashMap<>();
172         second.put(key2, value2);
173         Map<?, ?> combined = combine(first, second);
174 
175         assertNull(combined.remove(key1));
176         assertNull(combined.remove(nonExistingKey));
177         combined.clear();
178         assertThat(combined.size(), is(0));
179 
180         assertThat(first.get(key1), is(value1));
181         assertThat(first.size(), is(1));
182         assertThat(second.get(key2), is(value2));
183         assertThat(second.size(), is(1));
184 
185 
186         Object x = "x";
187         Object y = "y";
188         Object z = "z";
189         assumeThat(first, not(hasKey(x)));
190         assumeThat(second, not(hasKey(y)));
191         first.put(x, y);
192         second.put(y, z);
193         assertThat(combined.size(), is(1));
194         assertThat(first.size(), is(2));
195         assertThat(second.size(), is(2));
196 
197         combined.clear();
198         assertThat(combined.size(), is(0));
199         assertThat(first.size(), is(1));
200         assertThat(second.size(), is(1));
201     }

204     private <K, I, V> Map<K, V> combinedMap(K key, I internalValue, I internalKey, V value) {
205         Map<K, I> map1 = new HashMap<>();
206         Map<I, V> map2 = new HashMap<>();
207         map1.put(key, internalValue);
208         map2.put(internalKey, value);
209         return Maps.combine(map1, map2);
210     }
