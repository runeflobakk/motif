13  
14      @Test
15      public void doesNotInvokeImplementorOnNull() {
16          Fn<String, String> fn = new NullIfArgIsNull<String, String>() {
17              @Override
18              protected String orElse(String value) {
19                  fail("should not invoke this method");
20                  return "";
21              }
22          };
23          assertThat(fn.$(null), nullValue());
24      }
25  
26      @Test
27      public void passesFunctionArgToImplemtor() {
28          Fn<String, Integer> fn = new NullIfArgIsNull<String, Integer>() {
29              @Override
30              protected Integer orElse(String value) {
31                  return value.length();
32              }
33          };
34  
35          assertThat(fn.$("xy"), is(2));
36      }
37  
38      @Test
39      public void doesNotInvokeFn2ImplementorOnNullArgs() {
40          Fn2<Object, Object, Integer> fn2 = new NullIfEitherArgIsNull<Object, Object, Integer>() {
41              @Override protected Integer orElse(Object first, Object second) {
42                  fail("should not be called");
43                  return null;
44              }};
45  
46          assertThat(fn2.$(null, "a"), nullValue());
47          assertThat(fn2.$("a", null), nullValue());
48          assertThat(fn2.$(null, null), nullValue());
49      }
