50      public void bridgeArrayYieldingFunctionToIterableYieldingFunction() {
51          Fn<Object, Integer[]> oneTwoThree = new Fn<Object, Integer[]>() {
52              @Override
53              public Integer[] $(Object value) {
54                  return new Integer[] {1, 2, 3};
55              }
56          };
57          assertThat(optional("").split(Iterate.toIterable(oneTwoThree)), contains(1, 2, 3));
58      }

74      @Test
75      public void functionsAreAppliedOnEvalAndNeverAgain() {
76          @SuppressWarnings("unchecked")
77          Fn<String, String> register = mock(Fn.class);
78          when(register.$(anyString())).then(returnsFirstArg());
79  
80          Elements<String> elements = on(100, 200).map(first(toString).then(register));
81          verifyZeroInteractions(register);
82          elements = elements.eval();
83          verify(register).$("100");
84          verify(register).$("200");
85          verifyNoMoreInteractions(register);
86          assertThat(elements.map(before(1)), contains("1", "2"));
87          verifyNoMoreInteractions(register);
88      }
