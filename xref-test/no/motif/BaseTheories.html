22  @RunWith(Theories.class)
23  public class BaseTheories {
24  
25      @Theory
26      public void hashCode(@ForAll Object anyObject) {
27          assertThat(Base.hashCode.$(anyObject), is(Objects.hashCode(anyObject)));
28      }
29  
30      @Theory
31      public void containedInNullOrEmptyCollectionIsAlwaysFalse(@ForAll Object anyObject) {
32          assertFalse(containedIn(null).$(anyObject));
33          assertFalse(containedIn(emptyList()).$(anyObject));
34      }
35  
36      @Theory
37      public void containedInANonEmptyCollection(
38              @ForAll (sampleSize=20) Object inCollection,
39              @ForAll (sampleSize=20) Object anyOtherObject) {
40  
41          assumeThat(inCollection, not(anyOtherObject));
42          List<?> list = asList(inCollection);
43          assertTrue(containedIn(list).$(inCollection));
44          assertFalse(containedIn(list).$(anyOtherObject));
45      }
