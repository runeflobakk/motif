22  /**
23   * A test to cope with certain idioms which makes it impossible to achieve 100%
24   * test coverage.
25   */
26  public final class CoverageWorkaroundTest {
27  
28      private static final Logger LOG = LoggerFactory.getLogger(CoverageWorkaroundTest.class);
29  
30      private final Reflections motifPackage = new Reflections(Base.class.getPackage().getName(), new SubTypesScanner(false));
31      @Test
32      public void instantiateAllStaticUtilClassToAchieveCoverageForPrivateConstructors() throws Exception {
33          Set<Class<?>> allTypes = motifPackage.getSubTypesOf(Object.class);
34          for (Constructor<?> constructor : on(allTypes).filter(staticFinalUtilClass).map(declaredConstructor())) {
35              LOG.info("Instantiating " + constructor.getName());
36              assertThat(constructor.newInstance(), not(nullValue()));
37          }
38      }
39  
40  
41      static Predicate<Class<?>> staticFinalUtilClass = new Predicate<Class<?>>() {
42          @Override
43          public boolean $(Class<?> type) {
44              if (isFinal(type.getModifiers()) && type.getSuperclass() == Object.class) {
45                  Constructor<?>[] constructors = type.getDeclaredConstructors();
46                  if (constructors.length == 1 && constructors[0].getParameterTypes().length == 0) {
47                      return isPrivate(constructors[0].getModifiers());
48                  }
49              }
50              return false;
51          }
52      };
53  
54      static Fn<Class<?>, Constructor<?>> declaredConstructor(final Class<?> ... parameterTypes) {
55          return new Fn<Class<?>, Constructor<?>>() {
56              @Override
57              public Constructor<?> $(Class<?> type) {
58                  try {
59                      Constructor<?> constructor = type.getDeclaredConstructor(parameterTypes);
60                      constructor.setAccessible(true);
61                      return constructor;
62                  } catch (NoSuchMethodException | SecurityException e) {
63                      throw new RuntimeException(e.getMessage(), e);
64                  }
65              }
66          };
67      }
